SELECT bulletin.ID_bulletin, trimestre.Numero, etudiant.Nom, etudiant.Prenom, niveau.Nom, classe.Nom, bulletin.Appreciation
FROM bulletin
INNER JOIN 
trimestre
on bulletin.ID_trimestre=trimestre.ID_trimestre
INNER JOIN
inscription
on bulletin.ID_inscription=inscription.ID_inscription
INNER JOIN 
etudiant
on inscription.ID_etudiant=etudiant.ID_etudiant
INNER JOIN 
classe 
on inscription.ID_classe=classe.ID_classe
INNER JOIN
niveau
on classe.ID_niveau=niveau.ID_niveau
WHERE etudiant.Nom LIKE '%a%'
