SELECT trimestre.Numero, etudiant.Nom, etudiant.Prenom, discipline.Nom, AVG(evaluation.Note) AS Moyenne, detailbulletin.Appreciation
FROM detailbulletin
INNER JOIN 
enseignement
on detailbulletin.ID_enseignement=enseignement.ID_enseignement 
INNER JOIN 
discipline 
on enseignement.ID_discipline=discipline.ID_discipline 
INNER JOIN
bulletin
on detailbulletin.ID_bulletin=bulletin.ID_bulletin 
INNER JOIN 
inscription
on bulletin.ID_inscription=inscription.ID_inscription 
INNER JOIN 
etudiant 
on inscription.ID_etudiant=etudiant.ID_etudiant
INNER JOIN 
evaluation
on detailbulletin.ID_detailbulletin=evaluation.ID_detailbulletin
INNER JOIN 
trimestre
on trimestre.ID_trimestre = bulletin.ID_trimestre
GROUP BY discipline.ID_discipline, bulletin.ID_bulletin
HAVING bulletin.ID_bulletin=2001
