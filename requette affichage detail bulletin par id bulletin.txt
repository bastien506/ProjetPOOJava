SELECT etudiant.Nom, etudiant.Prenom, discipline.Nom, detailbulletin.Appreciation
FROM detailbulletin
INNER JOIN 
enseignement
on detailbulletin.ID_enseignement=enseignement.ID_enseignement 
INNER JOIN 
discipline 
on enseignement.ID_discipline=discipline.ID_discipline 
INNER JOIN
bulletin
on detailbulletin.ID_bulletin=bulletin.ID_bulletin 
INNER JOIN 
inscription
on bulletin.ID_inscription=inscription.ID_inscription 
INNER JOIN 
etudiant 
on inscription.ID_etudiant=etudiant.ID_etudiant
WHERE bulletin.ID_bulletin='2001'
